#include <stdio.h>

int main() {
      int lista1[] = {1, 2, 3, 4, 5}; // atribuição de lista
  int lista2[] = {2, 3, 4, 5, 6}; // atribuição de lista
  int tamanho = sizeof(lista1) / sizeof(lista1[0]); // operação matemática e atribuição
  int resultado[tamanho]; // atribuição (de tamanho)


  for (int i = 0; i < tamanho; i++) { // comparação, operação matemática e atribuição
    resultado[i] = lista1[i] * lista2[i]; // operação matemática e atribuição
  }


  printf("Resultado da multiplicação: ");
  for (int i = 0; i < tamanho; i++) { // comparação, operação matemática e atribuição
    printf("%d ", resultado[i]);
  }
  printf("\n");

  return 0; // atribuição (retorno de valor)
}
